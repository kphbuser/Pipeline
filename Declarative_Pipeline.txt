pipeline {
	agent any
	stages {
		stage('Build') {
			steps {
				sh 'java -version'
				sh 'echo "Welcome to declarative_Pipeline"'
				              		
			}
		}
		
		stage('Deploy'){
			steps {
				timeout(time: 3, unit: 'MINUTES')
					retry(4) {
						powershell '.\flakey-deploy.ps1'
					}
			}
		}
	}
}
